## Step 4: Quality Gate (2h)

Create `src/lib/coaching/quality-gate.ts`:

```typescript
export interface QualitySignals {
  messageCount: number;
  userSignaledSatisfaction: boolean;   // thumbs up, "thanks", "it works"
  hasCodeBlocks: boolean;              // response contained code
  hasCopyEvents: boolean;             // user copied from response
  conversationRating: number | null;   // 1-5 if rated
  totalResponseTokens: number;
}

export function shouldProcessConversation(signals: QualitySignals): boolean {
  // Multiple signals — any combination can trigger
  if (signals.messageCount >= 6) return true;
  if (signals.hasCodeBlocks && signals.hasCopyEvents) return true;
  if (signals.userSignaledSatisfaction && signals.messageCount >= 3) return true;
  if (signals.conversationRating !== null && signals.conversationRating >= 4) return true;
  if (signals.totalResponseTokens > 2000) return true;
  return false;
}

export function detectSatisfactionSignal(message: string): boolean {
  const patterns = [
    /\bthanks?\b/i, /\bthank you\b/i, /\bperfect\b/i,
    /\bit works?\b/i, /\bgot it\b/i, /\bgreat\b/i,
    /\bawesome\b/i, /\bexactly\b/i, /👍/, /🎉/,
  ];
  return patterns.some(p => p.test(message));
}
```

## Step 5: Embedding on Save Hooks (3h)

### 5a. Vault Save Hook

Create `src/lib/coaching/embed-on-save.ts`:

```typescript
import { generateEmbedding } from '@/lib/providers/embedding';

export async function embedVaultEntry(
  env: Env,
  entryId: string,
  content: string,
  userId: string
): Promise<string> {
  const { vectors } = await generateEmbedding(env, content);
  
  const vectorId = `vault_${entryId}`;
  await env.VECTORIZE.insert([{
    id: vectorId,
    values: vectors[0],
    metadata: {
      userId,
      type: 'vault_entry',
      entryId,
      title: content.slice(0, 100),
    },
  }]);
  
  // Update the entry with the vector ID
  await db.update(promptLibrary)
    .set({ embeddingVectorId: vectorId })
    .where(eq(promptLibrary.id, entryId));
  
  return vectorId;
}
```

Integrate into existing vault save routes:
- `src/app/api/vault/prompts/route.ts` POST handler — after successful insert, call `embedVaultEntry()`
- `src/app/api/vault/quick-save/route.ts` POST handler — same

```typescript
import { runWorkersAI } from '@/lib/providers/workers-ai';

export async function extractKnowledgeTags(env: Env, content: string): Promise<string[]> {
  const result = await runWorkersAI(env, {
    model: '@cf/ibm-granite/granite-4.0-h-micro',
    maxTokens: 100,
    temperature: 0.1,
  }, [{
    role: 'system',
    content: 'Extract 3-5 topic tags from this text. Return ONLY a JSON array of lowercase strings. No explanation.',
  }, {
    role: 'user',
    content: content.slice(0, 500),
  }]);
  
  try {
    return JSON.parse(result.content);
  } catch {
    return [];
  }
}
```
